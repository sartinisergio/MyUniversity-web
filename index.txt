  import * as duckdb from 'https://cdn.jsdelivr.net/npm/@duckdb/duckdb-wasm@1.28.0/+esm';

  // ── Stato globale ──────────────────────────────────────────
  let db, conn;
  let ultimiRisultati = [];
  let choicesMap = {};
  let lookupClassi = {};

  // Chiave localStorage per le modifiche
  const LS_KEY = 'myuniversity_modifiche_v1';

  // Oggetto corrente nel modal edit
  let editRecordCorrente = null;

  // ── Helpers localStorage ───────────────────────────────────
  function leggiModifiche() {
    try { return JSON.parse(localStorage.getItem(LS_KEY) || '{}'); }
    catch(e) { return {}; }
  }

  function salvaModifiche(obj) {
    localStorage.setItem(LS_KEY, JSON.stringify(obj));
  }

  /**
   * Genera una chiave univoca per il record.
   * Usiamo ateneo + laurea + insegnamento + anno_ins come chiave composita.
   */
  function chiaveRecord(row) {
    return [row.ateneo, row.laurea, row.insegnamento, row.anno_ins]
      .map(s => (s||'').trim())
      .join('||');
  }

  // Calcola anno accademico corrente (es. 2025/26)
  function annoAccademicoCorrente() {
    const oggi = new Date();
    const anno = oggi.getMonth() >= 8 ? oggi.getFullYear() : oggi.getFullYear() - 1;
    return `${anno}/${String(anno + 1).slice(-2)}`;
  }

  function dataOggi() {
    const d = new Date();
    const pad = n => String(n).padStart(2,'0');
    return `${pad(d.getDate())}/${pad(d.getMonth()+1)}/${d.getFullYear()} ${pad(d.getHours())}:${pad(d.getMinutes())}`;
  }

  // ── Applica modifiche locali a un array di righe ───────────
  function applicaModifiche(rows) {
    const mods = leggiModifiche();
    return rows.map(row => {
      const k = chiaveRecord(row);
      if (mods[k]) {
        return Object.assign({}, row, mods[k], { _modificato: true });
      }
      return row;
    });
  }

  // ── Inizializzazione DuckDB ────────────────────────────────
  async function initDuckDB() {
    const JSDELIVR_BUNDLES = duckdb.getJsDelivrBundles();
    const bundle = await duckdb.selectBundle(JSDELIVR_BUNDLES);
    const worker_url = URL.createObjectURL(
      new Blob([`importScripts("${bundle.mainWorker}");`], { type: 'text/javascript' })
    );
    const worker = new Worker(worker_url);
    const logger = new duckdb.ConsoleLogger();
    db = new duckdb.AsyncDuckDB(logger, worker);
    await db.instantiate(bundle.mainModule, bundle.pthreadWorker);
    conn = await db.connect();
    document.getElementById('status').textContent = 'Caricamento CSV...';
    await caricaCSV();
  }

  // ── Caricamento CSV ────────────────────────────────────────
  async function caricaCSV() {
    const [respIns, respLau] = await Promise.all([
      fetch('insegnamenti.csv'),
      fetch('lauree.csv')
    ]);
    const [csvIns, csvLau] = await Promise.all([respIns.text(), respLau.text()]);
    await db.registerFileText('insegnamenti.csv', csvIns);
    await db.registerFileText('lauree.csv', csvLau);

    await conn.query(`
      CREATE TABLE insegnamenti AS
      SELECT * FROM read_csv_auto('insegnamenti.csv', header=true, ignore_errors=true);
    `);
    await conn.query(`
      CREATE TABLE lauree AS
      SELECT * FROM read_csv_auto('lauree.csv', header=true, ignore_errors=true);
    `);

    document.getElementById('status').textContent = 'Pronto';
    document.getElementById('tabella-container').innerHTML =
      '<table id="tabella" class="display" style="width:100%"></table>';

    await popolaFiltri();
    inizializzaTabella();
    inizializzaChoices();
    document.getElementById('btn-export').disabled = false;
  }

  // ── Popola dropdown ────────────────────────────────────────
  async function popolaFiltri() {
    try {
      const rispLookup = await fetch('classi_lookup.json');
      lookupClassi = await rispLookup.json();
    } catch(e) { lookupClassi = {}; }

    const fill = async (selectId, col, tabella) => {
      const res = await conn.query(
        `SELECT DISTINCT ${col} FROM ${tabella} WHERE ${col} IS NOT NULL ORDER BY ${col}`
      );
      const sel = document.getElementById(selectId);
      res.toArray().forEach(r => {
        const v = String(Object.values(r)[0]);
        if (v && v !== 'null') {
          const opt = document.createElement('option');
          opt.value = v; opt.textContent = v;
          sel.appendChild(opt);
        }
      });
    };

    const aree = await conn.query(`
      SELECT DISTINCT regexp_extract(ssd, '^([A-Z]+)/', 1) AS area
      FROM insegnamenti WHERE ssd IS NOT NULL AND ssd != '' ORDER BY area
    `);
    const selArea = document.getElementById('f-area');
    aree.toArray().forEach(r => {
      const v = String(r.area);
      if (v && v !== 'null' && v !== '') {
        const opt = document.createElement('option');
        opt.value = v; opt.textContent = v;
        selArea.appendChild(opt);
      }
    });

    const ssdCodici = await conn.query(`
      SELECT DISTINCT regexp_extract(ssd, '^([^ ]+)', 1) AS codice
      FROM insegnamenti WHERE ssd IS NOT NULL AND ssd != '' ORDER BY codice
    `);
    const selSsd = document.getElementById('f-ssd');
    ssdCodici.toArray().forEach(r => {
      const v = String(r.codice);
      if (v && v !== 'null' && v !== '') {
        const opt = document.createElement('option');
        opt.value = v; opt.textContent = v;
        selSsd.appendChild(opt);
      }
    });

    await fill('f-zona',      'zona',     'insegnamenti');
    await fill('f-regione',   'regione',  'insegnamenti');
    await fill('f-ateneo',    'ateneo',   'insegnamenti');
    await fill('f-tipologia', 'tipologia','insegnamenti');
    await popolaClassi(null);
  }

  async function popolaClassi(filtroSsd, filtroTipologia = []) {
    const conditions = [];
    if (filtroSsd && filtroSsd.length > 0) {
      const listaSsd = filtroSsd.map(s => `'${s.replace(/'/g,"''")}'`).join(',');
      conditions.push(`regexp_extract(ssd, '^([^ ]+)', 1) IN (${listaSsd})`);
    }
    if (filtroTipologia && filtroTipologia.length > 0) {
      const listaTip = filtroTipologia.map(t => `'${t.replace(/'/g,"''")}'`).join(',');
      conditions.push(`tipologia IN (${listaTip})`);
    }
    const whereClause = conditions.length > 0
      ? `WHERE nome_classe IS NOT NULL AND ${conditions.join(' AND ')}`
      : `WHERE nome_classe IS NOT NULL`;
    const sql = `SELECT DISTINCT nome_classe FROM insegnamenti ${whereClause}`;
    const classiRes = await conn.query(sql);

    function ordinaCodice(codice) {
      if (!codice) return 'ZZZ';
      const prefissi = {'L-':'1','L/':'2','LM-':'3','LM/':'4','LMG':'5','LMR':'6'};
      let prefisso = '9';
      for (const [p, v] of Object.entries(prefissi)) {
        if (codice.startsWith(p)) { prefisso = v; break; }
      }
      const num = parseInt(codice.replace(/[^0-9]/g, '')) || 999;
      return prefisso + String(num).padStart(4, '0');
    }

    const voci = [];
    classiRes.toArray().forEach(r => {
      const nome = String(r.nome_classe);
      if (nome && nome !== 'null') {
        const codice = lookupClassi[nome.toLowerCase()] || '';
        voci.push({ nome, codice, sort: ordinaCodice(codice) });
      }
    });
    voci.sort((a, b) => a.sort.localeCompare(b.sort));

    const ch = choicesMap['f-classe'];
    if (ch) {
      ch.clearChoices();
      ch.setChoices(
        voci.map(({ nome, codice }) => ({
          value: nome,
          label: codice ? `${codice} · ${nome}` : nome,
        })),
        'value', 'label', true
      );
    } else {
      const sel = document.getElementById('f-classe');
      sel.innerHTML = '';
      voci.forEach(({ nome, codice }) => {
        const opt = document.createElement('option');
        opt.value = nome;
        opt.textContent = codice ? `${codice} · ${nome}` : nome;
        sel.appendChild(opt);
      });
    }
  }

  async function aggiornaCascataClassi() {
    if (!conn) return;
    const ssdSel  = choicesMap['f-ssd']  ? choicesMap['f-ssd'].getValue(true)  : [];
    const areaSel = choicesMap['f-area'] ? choicesMap['f-area'].getValue(true) : [];
    const tipSel  = choicesMap['f-tipologia'] ? choicesMap['f-tipologia'].getValue(true) : [];

    let tuttiSsd = [...ssdSel];
    if (areaSel.length > 0) {
      const listaAree = areaSel.map(a => `'${a.replace(/'/g,"''")}'`).join(',');
      const res = await conn.query(`
        SELECT DISTINCT regexp_extract(ssd, '^([^ ]+)', 1) AS codice
        FROM insegnamenti
        WHERE regexp_extract(ssd, '^([A-Z]+)/', 1) IN (${listaAree})
          AND ssd IS NOT NULL AND ssd != ''
      `);
      res.toArray().forEach(r => {
        const v = String(r.codice);
        if (v && v !== 'null' && !tuttiSsd.includes(v)) tuttiSsd.push(v);
      });
    }

    const haSelezione = tuttiSsd.length > 0 || tipSel.length > 0;
    await popolaClassi(tuttiSsd.length > 0 ? tuttiSsd : null, tipSel);

    const badge  = document.getElementById('cascata-badge');
    const gruppo = document.getElementById('f-classe').closest('.filtro-gruppo');
    badge.style.display  = haSelezione ? 'inline-block' : 'none';
    badge.textContent    = [
      tuttiSsd.length > 0 ? 'filtrata per SSD' : '',
      tipSel.length > 0   ? 'filtrata per tipologia' : ''
    ].filter(Boolean).join(' · ');
    gruppo.classList.toggle('cascata-attiva', haSelezione);
  }

  // ── buildWhere ─────────────────────────────────────────────
  function getVals(id) {
    const ch = choicesMap[id];
    if (ch) return ch.getValue(true);
    const el = document.getElementById(id);
    return el ? [el.value].filter(Boolean) : [];
  }

  function inClause(col, vals) {
    if (!vals || vals.length === 0) return null;
    if (vals.length === 1) return `${col} = '${vals[0].replace(/'/g,"''")}'`;
    return `${col} IN (${vals.map(v => `'${v.replace(/'/g,"''")}'`).join(',')})`;
  }

  function buildWhere() {
    const clauses = [];
    const ins = document.getElementById('f-insegnamento').value.trim();
    if (ins) {
      const gruppiOR = ins.split(',').map(g => g.trim()).filter(g => g.length > 0);
      const clausoleOR = gruppiOR.map(gruppo => {
        const paroleAND = gruppo.split(/\s+/).map(p => p.trim()).filter(p => p.length > 0);
        if (paroleAND.length === 1) {
          return `lower(i.insegnamento) LIKE '%${paroleAND[0].toLowerCase().replace(/'/g,"''")}%'`;
        } else {
          return '(' + paroleAND
            .map(p => `lower(i.insegnamento) LIKE '%${p.toLowerCase().replace(/'/g,"''")}%'`)
            .join(' AND ') + ')';
        }
      });
      clauses.push(clausoleOR.length === 1 ? clausoleOR[0] : '(' + clausoleOR.join(' OR ') + ')');
    }
    const area = getVals('f-area');
    if (area.length) clauses.push(
      `regexp_extract(i.ssd, '^([A-Z]+)/', 1) IN (${area.map(a=>`'${a.replace(/'/g,"''")}'`).join(',')})`
    );
    const ssd = getVals('f-ssd');
    if (ssd.length) clauses.push(
      `regexp_extract(i.ssd, '^([^ ]+)', 1) IN (${ssd.map(s=>`'${s.replace(/'/g,"''")}'`).join(',')})`
    );
    const zona = getVals('f-zona');
    const c1 = inClause('i.zona', zona); if (c1) clauses.push(c1);
    const reg = getVals('f-regione');
    const c2 = inClause('i.regione', reg); if (c2) clauses.push(c2);
    const ate = getVals('f-ateneo');
    const c3 = inClause('i.ateneo', ate); if (c3) clauses.push(c3);
    const tip = getVals('f-tipologia');
    const c4 = inClause('i.tipologia', tip); if (c4) clauses.push(c4);
    const cls = getVals('f-classe');
    const c5 = inClause('i.nome_classe', cls); if (c5) clauses.push(c5);
    const anno = getVals('f-anno');
    if (anno.length) clauses.push(`i.anno_ins IN (${anno.join(',')})`);
    return clauses.length ? 'WHERE ' + clauses.join(' AND ') : '';
  }

  function escHtml(s) {
    return (s||'').replace(/&/g,'&amp;').replace(/"/g,'&quot;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
  }

  // ── Ricerca ────────────────────────────────────────────────
  async function eseguiRicerca() {
    if (!conn) return;
    document.getElementById('status').textContent = 'Ricerca in corso...';
    const where = buildWhere();
    const sql = `
      SELECT
        i.insegnamento, i.ssd, i.ateneo, i.regione, i.zona,
        i.laurea, i.nome_classe, i.tipologia, i.durata,
        CAST(i.anno_ins AS VARCHAR) AS anno_ins,
        CAST(i.cfu AS VARCHAR) AS cfu,
        CAST(i.n_curriculum AS VARCHAR) AS n_curriculum,
        COALESCE(i.curriculum_lista, '') AS curriculum_lista,
        i.trasversale,
        COALESCE(i.url_ins, '') AS url_ins
      FROM insegnamenti i
      ${where}
      ORDER BY i.ateneo, i.laurea, i.insegnamento
      LIMIT 5000
    `;
    try {
      const result = await conn.query(sql);
      let rows = result.toArray().map(r => {
        const obj = {};
        Object.keys(r).forEach(k => { obj[k] = r[k] === null ? '' : String(r[k]); });
        return obj;
      });

      // Applica modifiche locali
      rows = applicaModifiche(rows);

      ultimiRisultati = rows;
      dtTable.clear().rows.add(rows).draw();

      const totRes = await conn.query(`SELECT COUNT(*) AS n FROM insegnamenti i ${where}`);
      const tot = totRes.toArray()[0].n;
      document.getElementById('contatore').innerHTML =
        `<strong>${Number(tot).toLocaleString('it-IT')}</strong> insegnamenti trovati` +
        (tot > 5000 ? ' (visualizzati i primi 5.000)' : '');
      document.getElementById('status').textContent = 'Pronto';
    } catch(e) {
      console.error(e);
      document.getElementById('status').textContent = 'Errore nella query';
    }
  }

  // ── Export CSV ─────────────────────────────────────────────
  function esportaCSV() {
    if (!ultimiRisultati.length) {
      alert('Nessun dato da esportare. Esegui prima una ricerca.');
      return;
    }
    const colonne = [
      'insegnamento','ssd','ateneo','regione','zona',
      'laurea','nome_classe','tipologia','durata',
      'anno_ins','cfu','n_curriculum','curriculum_lista',
      'trasversale','url_ins','anno_accademico_agg','data_aggiornamento','note_agg'
    ];
    const intestazione = colonne.join(';');
    const righe = ultimiRisultati.map(row =>
      colonne.map(col => {
        const v = row[col] ?? '';
        return (String(v).includes(';') || String(v).includes('"') || String(v).includes('\n'))
          ? `"${String(v).replace(/"/g, '""')}"` : String(v);
      }).join(';')
    );
    const csv = '\uFEFF' + [intestazione, ...righe].join('\r\n');
    const blob = new Blob([csv], { type: 'text/csv;charset=utf-8;' });
    const url  = URL.createObjectURL(blob);
    const a    = document.createElement('a');
    a.href = url;
    a.download = `myuniversity_export_${new Date().toISOString().slice(0,10)}.csv`;
    document.body.appendChild(a);
    a.click();
    document.body.removeChild(a);
    URL.revokeObjectURL(url);
  }

  // ── MODALE MODIFICA ────────────────────────────────────────
  function apriModalEdit(row) {
    editRecordCorrente = row;
    const mods = leggiModifiche();
    const k = chiaveRecord(row);
    const mod = mods[k] || {};

    document.getElementById('edit-sottotitolo').textContent =
      `${row.ateneo} · ${row.laurea}`;

    // Record originale (dati MUR)
    document.getElementById('edit-orig-text').textContent =
      ` SSD: ${row.ssd} · CFU: ${row.cfu} · Anno: ${row.anno_ins} · URL: ${row.url_ins || 'N/A'}`;

    // Popola form con valore modificato oppure originale
    document.getElementById('edit-insegnamento').value = mod.insegnamento ?? row.insegnamento ?? '';
    document.getElementById('edit-ssd').value          = mod.ssd ?? row.ssd ?? '';
    document.getElementById('edit-cfu').value          = mod.cfu ?? row.cfu ?? '';
    document.getElementById('edit-anno').value         = mod.anno_ins ?? row.anno_ins ?? '1';

  async function mostraPianoDiStudi(ateneo, laurea, tipologia) {
    const overlay = document.getElementById('modal-overlay');
    const body    = document.getElementById('modal-body');
    document.getElementById('modal-titolo').textContent = laurea || 'Piano di studi';
    document.getElementById('modal-sottotitolo').textContent = `${ateneo} · ${tipologia || ''}`;
    body.innerHTML = '<div class="loading">⏳ Caricamento piano di studi...</div>';
    overlay.classList.add('aperto');

    try {
      const sql = `
        SELECT
          CAST(anno_ins AS INTEGER) AS anno_ins,
          insegnamento, ssd,
          CAST(MAX(cfu) AS VARCHAR) AS cfu,
          STRING_AGG(DISTINCT curriculum_lista, ' · ') AS curriculum_lista,
          MAX(COALESCE(trasversale, '')) AS trasversale,
          MAX(COALESCE(url_ins, '')) AS url_ins
        FROM insegnamenti
        WHERE ateneo = '${ateneo.replace(/'/g,"''")}' 
          AND laurea = '${laurea.replace(/'/g,"''")}' 
        GROUP BY anno_ins, insegnamento, ssd
        ORDER BY anno_ins, insegnamento
      `;
      const res  = await conn.query(sql);
      let rows = res.toArray().map(r => {
        const obj = {};
        Object.keys(r).forEach(k => { obj[k] = r[k] === null ? '' : String(r[k]); });
        return obj;
      });

      // Applica modifiche locali anche nel piano di studi
      rows = applicaModifiche(rows);

      if (!rows.length) {
        body.innerHTML = '<p style="color:#999;padding:20px">Nessun insegnamento trovato.</p>';
        return;
      }

      const perAnno = {};
      rows.forEach(r => {
        const a = r.anno_ins || '?';
        if (!perAnno[a]) perAnno[a] = [];
        perAnno[a].push(r);
      });

      let html = '';
      const etichette = {'1':'1° anno','2':'2° anno','3':'3° anno',
                         '4':'4° anno','5':'5° anno','6':'6° anno','?':'Anno non specificato'};

      Object.keys(perAnno).sort().forEach(anno => {
        const ins = perAnno[anno];
        const cfuTot = ins.reduce((s, r) => s + (parseFloat(r.cfu) || 0), 0);
        html += `<div class="anno-block">
          <h4>${etichette[anno] || anno + '° anno'} — ${ins.length} insegnamenti · ${cfuTot} CFU</h4>
          <table class="piano-table">
            <thead><tr>
              <th>Insegnamento</th><th>SSD</th><th>CFU</th><th>Curriculum</th><th>Link</th>
            </tr></thead>
            <tbody>`;
        ins.forEach(r => {
          const trasv = (r.trasversale === '1' || r.trasversale === 'true')
            ? '<span class="badge-trasv">trasversale</span>' : '';
          const modIcn = r._modificato
            ? '<span title="Modificato localmente" style="margin-left:4px">🔄</span>' : '';
          const link = r.url_ins && r.url_ins !== 'NA' && r.url_ins !== 'null' && r.url_ins !== ''
            ? `<a href="${escHtml(r.url_ins)}" target="_blank">🔗</a>` : '';
          html += `<tr ${r._modificato ? 'style="background:#fffbe6"' : ''}>
            <td>${escHtml(r.insegnamento)}${trasv}${modIcn}</td>
            <td>${escHtml(r.ssd)}</td>
            <td>${escHtml(r.cfu)}</td>
            <td>${escHtml(r.curriculum_lista)}</td>
            <td>${link}</td>
          </tr>`;
        });
        html += `</tbody></table></div>`;
      });

      const totIns = rows.length;
      const totCfu = rows.reduce((s, r) => s + (parseFloat(r.cfu) || 0), 0);
      html = `<div style="background:#fde8ec;border-radius:8px;padding:12px 16px;
                          margin-bottom:20px;font-size:0.85rem;color:#c8102e;font-weight:600;">
        📊 <strong>${totIns}</strong> insegnamenti · <strong>${totCfu}</strong> CFU complessivi
      </div>` + html;

      body.innerHTML = html;
    } catch(e) {
      console.error(e);
      body.innerHTML = '<p style="color:red;padding:20px">Errore nel caricamento.</p>';
    }
  }

  let alberoCompleto = [];   // struttura dati per la vista ateneo
  function ordinaCodiceClasse(nomeClasse) {
  const codice = lookupClassi[(nomeClasse || '').toLowerCase()] || '';
  if (!codice) return 99999;
  const pesi = {'L-':1,'L/':2,'LM-':3,'LM/':4,'LMG':5,'LMR':6,'LS-':7};
  let peso = 80000;
  for (const [p, v] of Object.entries(pesi)) {
    if (codice.startsWith(p)) { peso = v * 10000; break; }
  }
  return peso + (parseInt(codice.replace(/[^0-9]/g,''), 10) || 0);
}
  async function costruisciVistaAteneo() {
    if (!conn) return;
    // Ricarica lookup se vuoto
    if (Object.keys(lookupClassi).length === 0) {
    try {
    const r = await fetch('classi_lookup.json');
    lookupClassi = await r.json();
  } catch(e) { lookupClassi = {}; }
}
    document.getElementById('va-status').textContent = 'Caricamento struttura...';
    document.getElementById('va-tree-container').innerHTML =
      '<div class="loading">⏳ Elaborazione struttura ad albero...</div>';

    try {
      const sql = `
        SELECT
          ateneo,
          nome_classe,
          tipologia,
          laurea,
          COUNT(*) AS n_ins,
          SUM(TRY_CAST(cfu AS DOUBLE)) AS tot_cfu
        FROM insegnamenti
        WHERE ateneo IS NOT NULL AND laurea IS NOT NULL
        GROUP BY ateneo, nome_classe, tipologia, laurea
        ORDER BY ateneo, nome_classe, laurea
      `;
      const res = await conn.query(sql);
      const rows = res.toArray().map(r => ({
        ateneo:      String(r.ateneo || ''),
        nome_classe: String(r.nome_classe || 'Classe non specificata'),
        tipologia:   String(r.tipologia || ''),
        laurea:      String(r.laurea || ''),
        n_ins:       Number(r.n_ins || 0),
        tot_cfu:     Math.round(Number(r.tot_cfu || 0)),
      }));

      // Costruisce struttura nested
      // Livello 1: ateneo
      // Livello 2: nome_classe
      // Livello 3: laurea (corso singolo)
      const map = {};
      rows.forEach(r => {
        if (!map[r.ateneo]) map[r.ateneo] = { classi: {}, n_ins: 0, tot_cfu: 0 };
        map[r.ateneo].n_ins  += r.n_ins;
        map[r.ateneo].tot_cfu += r.tot_cfu;

        const cls = r.nome_classe;
        if (!map[r.ateneo].classi[cls]) map[r.ateneo].classi[cls] = { corsi: [], n_ins: 0, tot_cfu: 0 };
        map[r.ateneo].classi[cls].n_ins  += r.n_ins;
        map[r.ateneo].classi[cls].tot_cfu += r.tot_cfu;
        map[r.ateneo].classi[cls].corsi.push({
          laurea:    r.laurea,
          tipologia: r.tipologia,
          n_ins:     r.n_ins,
          tot_cfu:   r.tot_cfu,
        });
      });

      alberoCompleto = Object.entries(map)
        .sort((a, b) => a[0].localeCompare(b[0]))
        .map(([nomeAteneo, datiAteneo]) => ({
          ateneo:   nomeAteneo,
          n_ins:    datiAteneo.n_ins,
          tot_cfu:  datiAteneo.tot_cfu,
          classi: Object.entries(datiAteneo.classi)
          .sort((a, b) => ordinaCodiceClasse(a[0]) - ordinaCodiceClasse(b[0]))
          .map(([nomeClasse, datiClasse]) => ({
          classe:  nomeClasse,
              n_ins:   datiClasse.n_ins,
              tot_cfu: datiClasse.tot_cfu,
              corsi:   datiClasse.corsi.sort((a, b) => a.laurea.localeCompare(b.laurea)),
            }))
          }));

      renderAlbero(alberoCompleto);
      document.getElementById('va-contatore').textContent =
        `${alberoCompleto.length} atenei · ${rows.length} corsi di laurea`;
      document.getElementById('va-status').textContent = 'Pronto';
    } catch(e) {
      console.error(e);
      document.getElementById('va-status').textContent = 'Errore';
      document.getElementById('va-tree-container').innerHTML =
        '<p style="color:red;padding:20px">Errore nel caricamento della struttura.</p>';
    }
  }

  function renderAlbero(dati) {
    const container = document.getElementById('va-tree-container');
    if (!dati.length) {
      container.innerHTML = '<p style="color:#999;padding:20px">Nessun dato.</p>';
      return;
    }

    const tipoBadgeClass = t => {
      const m = { 'triennale':'badge-triennale', 'magistrale':'badge-magistrale',
                  'ciclo unico':'badge-ciclo_unico' };
      return m[(t||'').toLowerCase()] || 'badge-triennale';
    };

    let html = '<div class="tree">';
    dati.forEach(a => {
      html += `
        <div class="tree-ateneo tree-node" data-ateneo="${escHtml(a.ateneo)}">
          <div class="tree-ateneo-header" onclick="toggleNodo(this)">
            <span class="freccia">▶</span>
            <span>${escHtml(a.ateneo)}</span>
            <span class="badge-stat">
              ${a.n_ins.toLocaleString('it-IT')} ins. · ${a.tot_cfu.toLocaleString('it-IT')} CFU
            </span>
          </div>
          <div class="tree-ateneo-children">`;
      a.classi.forEach(cl => {
        html += `
            <div class="tree-classe tree-node">
              <div class="tree-classe-header" onclick="toggleNodo(this)">
                <span class="freccia">▶</span>
                <span>${lookupClassi[(cl.classe||'').toLowerCase()] ? lookupClassi[(cl.classe||'').toLowerCase()] + ' · ' : ''}${escHtml(cl.classe)}</span>
                <span class="badge-stat-classe">
                  ${cl.n_ins} ins. · ${cl.tot_cfu} CFU
                </span>
              </div>
              <div class="tree-classe-children">`;
        cl.corsi.forEach(corso => {
          html += `
                <div class="tree-corso"
                     onclick="apriPianoFromTree('${escHtml(a.ateneo).replace(/'/g,"\\'")}','${escHtml(corso.laurea).replace(/'/g,"\\'")}','${escHtml(corso.tipologia).replace(/'/g,"\\'")}')">
                  <span class="badge badge-tipo ${tipoBadgeClass(corso.tipologia)}">${escHtml(corso.tipologia)}</span>
                  <span>${escHtml(corso.laurea)}</span>
                  <span class="stat-corso">${corso.n_ins} ins. · ${corso.tot_cfu} CFU</span>
                </div>`;
        });
        html += `</div></div>`; // chiude tree-classe-children e tree-classe
      });
      html += `</div></div>`; // chiude tree-ateneo-children e tree-ateneo
    });
    html += '</div>';
    container.innerHTML = html;
  }

  document.getElementById('va-search').addEventListener('input', function() {
    const q = this.value.trim().toLowerCase();
    if (!q) { renderAlbero(alberoCompleto); return; }

    const filtrato = alberoCompleto
      .map(a => {
        if (a.ateneo.toLowerCase().includes(q)) return a; // ateneo match → mostra tutto
        const classiFiltrate = a.classi
          .map(cl => {
            const corsiFiltrati = cl.corsi.filter(c =>
              c.laurea.toLowerCase().includes(q) ||
              cl.classe.toLowerCase().includes(q)
            );
            if (!corsiFiltrati.length) return null;
            return { ...cl, corsi: corsiFiltrati,
              n_ins: corsiFiltrati.reduce((s,c)=>s+c.n_ins,0),
              tot_cfu: corsiFiltrati.reduce((s,c)=>s+c.tot_cfu,0) };
          })
          .filter(Boolean);
        if (!classiFiltrate.length) return null;
        return { ...a, classi: classiFiltrate,
          n_ins: classiFiltrate.reduce((s,cl)=>s+cl.n_ins,0),
          tot_cfu: classiFiltrate.reduce((s,cl)=>s+cl.tot_cfu,0) };
      })
      .filter(Boolean);

    renderAlbero(filtrato);

    // Apri automaticamente tutti i nodi quando si cerca
    if (q) {
      document.querySelectorAll('.tree-ateneo, .tree-classe').forEach(n => n.classList.add('aperto'));
    }
  });
